49 inline bool separateMqttPacketByteBuffers(const ByteBuffer& buffer, std::vector<ByteBuffer>& packets, std::size_t& leftOverPosition)
50 {
51 packets.clear();
52 leftOverPosition = 0;
53
54 size_t packetStart{ 0 };
55 while (packetStart < buffer.size())
56 {
57 // If there are not enough bytes left to read a full packet header, break out of the loop.
58 if (buffer.size() - buffer.readCursor() < 2)
59 {
60 return !packets.empty();;
61 }
62
63 buffer.resetReadCursor();
64 assert(buffer.readCursor() == 0);
65
66 buffer.incrementReadCursor(packetStart + 1);//Ignore the first byte (packet type and flags).
67 mqtt::VariableByteInteger remainingLength{ mqtt::VariableByteInteger::tryCreateFromBuffer(buffer) };
68
69 //Full packet size = Remaining Length + 1 byte Fixed Header + Remaining Length field size.
70 size_t packetSize{ remainingLength.uint32Value() + sizeof(std::uint8_t) + remainingLength.encodingSize() };
71
72 if (buffer.size() - packetStart < packetSize)
73 {
74 //Not enough bytes for the full packet.
75 leftOverPosition = buffer.size() - packetStart;
76 return !packets.empty();
77 }
78
79 ByteBuffer packetBuffer(packetSize);
80
81 try
82 {
83 packetBuffer.append(buffer.bytes() + packetStart, packetSize);
84 }
85 catch (const std::exception& e)
86 {
87 LogError("Utils", "separateMqttPacketByteBuffers(), Failed to separate MQTT packets: %s", e.what());
88 return false;
89 }
90
91 packets.push_back(std::move(packetBuffer));
92
93 packetStart += packetSize;
94 }
95
96 return !packets.empty();
97 }
